name: Auto Check Lab

on:
  pull_request:
    branches: [main]

jobs:
  check:
    runs-on: ubuntu-latest
    permissions:
      pull-requests: write
    steps:
      - uses: actions/checkout@v4

      - name: Check files
        id: check
        run: |
          echo "has_report=false" >> $GITHUB_OUTPUT
          echo "has_code=false" >> $GITHUB_OUTPUT
          echo "has_ml=false" >> $GITHUB_OUTPUT
          echo "has_viz=false" >> $GITHUB_OUTPUT
          echo "has_competency=false" >> $GITHUB_OUTPUT

          if find . -name "report.md" -o -name "*.ipynb" | grep -q .; then
            echo "has_report=true" >> $GITHUB_OUTPUT
          fi

          if find . -name "*.py" | grep -q .; then
            echo "has_code=true" >> $GITHUB_OUTPUT
          fi

          if grep -rq "RandomForest\|LinearRegression\|keras\|torch" --include="*.py" --include="*.ipynb" . 2>/dev/null; then
            echo "has_ml=true" >> $GITHUB_OUTPUT
          fi

          if grep -rq "plotly\|matplotlib\|seaborn\|Scatter" --include="*.py" --include="*.ipynb" . 2>/dev/null; then
            echo "has_viz=true" >> $GITHUB_OUTPUT
          fi

          if grep -rqi "PK-\|competency\|indicator" --include="*.md" . 2>/dev/null; then
            echo "has_competency=true" >> $GITHUB_OUTPUT
          fi

      - name: Comment PR
        uses: peter-evans/create-or-update-comment@v4
        with:
          issue-number: ${{ github.event.pull_request.number }}
          body: |
            ## Auto Check Result

            Hello @${{ github.actor }}!

            | Check | Status |
            |-------|--------|
            | Report/Notebook | ${{ steps.check.outputs.has_report == 'true' && 'PASS' || 'FAIL' }} |
            | Code (.py) | ${{ steps.check.outputs.has_code == 'true' && 'PASS' || 'FAIL' }} |
            | ML Model | ${{ steps.check.outputs.has_ml == 'true' && 'PASS' || 'FAIL' }} |
            | Visualization | ${{ steps.check.outputs.has_viz == 'true' && 'PASS' || 'FAIL' }} |
            | Competency ref | ${{ steps.check.outputs.has_competency == 'true' && 'PASS' || 'FAIL' }} |

            *Generated by GitHub Actions for GenAI course*
